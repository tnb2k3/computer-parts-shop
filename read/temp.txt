# Reset Database Script\n# This script will drop and recreate the database with fresh data\n\nWrite-Host "========================================" -ForegroundColor Cyan\nWrite-Host "  Computer Shop - Database Reset" -ForegroundColor Cyan\nWrite-Host "========================================" -ForegroundColor Cyan\nWrite-Host ""\n\n# Check if Docker containers are running\nWrite-Host "Checking Docker containers..." -ForegroundColor Yellow\n$containerStatus = docker ps --filter "name=computer_shop_mysql" --format "{{.Status}}"\n\nif (-not $containerStatus) {\n    Write-Host "Error: MySQL container is not running!" -ForegroundColor Red\n    Write-Host "Please start Docker containers first with: docker-compose up -d" -ForegroundColor Yellow\n    exit 1\n}\n\nWrite-Host "MySQL container is running" -ForegroundColor Green\nWrite-Host ""\n\n# Confirm with user\nWrite-Host "WARNING: This will DELETE all existing data!" -ForegroundColor Red\n$confirmation = Read-Host "Are you sure you want to reset the database? (yes/no)"\n\nif ($confirmation -ne "yes") {\n    Write-Host "Operation cancelled." -ForegroundColor Yellow\n    exit 0\n}\n\nWrite-Host ""\nWrite-Host "Resetting database..." -ForegroundColor Yellow\nWrite-Host "Step 1: Copying schema file to container..." -ForegroundColor Cyan\n\n# Copy schema.sql to container\ndocker cp src/Database/schema.sql computer_shop_mysql:/tmp/schema.sql\n\nif ($LASTEXITCODE -ne 0) {\n    Write-Host "Error: Failed to copy schema file!" -ForegroundColor Red\n    exit 1\n}\n\nWrite-Host "Step 2: Importing database..." -ForegroundColor Cyan\n\n# Import schema.sql from inside the container\ndocker exec computer_shop_mysql mysql -uroot -proot --default-character-set=utf8mb4 -e "source /tmp/schema.sql"\n\nif ($LASTEXITCODE -eq 0) {\n    Write-Host ""\n    Write-Host "========================================" -ForegroundColor Green\n    Write-Host "  Database reset successful!" -ForegroundColor Green\n    Write-Host "========================================" -ForegroundColor Green\n    Write-Host ""\n    Write-Host "Database has been reset with:" -ForegroundColor Cyan\n    Write-Host "  - 10 Product categories" -ForegroundColor White\n    Write-Host "  - 15 Sample products with full image URLs" -ForegroundColor White\n    Write-Host "  - 2 Admin users (admin, admin_bao)" -ForegroundColor White\n    Write-Host "  - 1 Customer user" -ForegroundColor White\n    Write-Host ""\n    Write-Host "You can now access the application at: http://localhost:8080" -ForegroundColor Cyan\n}\nelse {\n    Write-Host ""\n    Write-Host "Error: Failed to reset database!" -ForegroundColor Red\n    Write-Host "Please check the error messages above." -ForegroundColor Yellow\n    exit 1\n}\n
